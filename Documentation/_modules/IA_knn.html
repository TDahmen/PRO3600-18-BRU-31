[docs]def adapt(config):
    res=[0 for i in range(64)]
    for k in range(12):
        for l in range(12):
            if (k>=2 and k<=9) and (l>=2 and l<=9):
                res[8*(k-2)+(l-2)]=echecs.piece_converter(config[k][l])
    return(np.array(res))
